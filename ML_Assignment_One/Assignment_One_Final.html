def calc_sse_reg(X, y, w, lamda):
    fst = 0
    for i in range(len(X)):
        sub = y[i] - np.dot(w.T, X[i])
        fst += np.dot(sub, sub)
    
    snd = lamda * (np.linalg.norm(w, 2) ** 2)
    return fst + snd
